CREATE TABLE WLAN_SENSOR_ACTIVE(
	ID VARCHAR(65) PRIMARY KEY,
	TIME INT NOT NULL,
	MAC VARCHAR(65) NOT NULL,
	IPV4 VARCHAR(65) NOT NULL,
	IPERF_DOWN INT(6),
	IPERF_UP INT(6),
	DHCP_TIME INT(12),
	DHCP_STATUS INT(1),
	SPEEDTEST_DOWN INT(6),
	SPEEDTEST_UP INT(6),
	SPEEDTEST_RTT INT(6),
	ICMP_UOL INT(6),
	ICMP_GOOGLE INT(6),
	ICMP_CISCO INT(6),
	ICMP_BBC INT(6),
	DNS_UOL INT(6),
	DNS_GOOGLE INT(6),
	DNS_CISCO INT(6),
	DNS_BBC INT(6),
	HTTP_UOL INT(6),
	HTTP_GOOGLE INT(6),
	HTTP_CISCO INT(6),
	HTTP_BBC INT(6)
);


CREATE TABLE WLAN_SENSOR_PASSIVE(
	ID VARCHAR(65) PRIMARY KEY,
	TIME INT NOT NULL,
	MAC VARCHAR(65) NOT NULL,
	IPV4 VARCHAR(65) NOT NULL,
	DATARATE INT(6) NOT NULL,
	RX_SIGNAL INT(6) NOT NULL,
	TX_POWER INT(6) NOT NULL,
	RETRIES INT(6) NOT NULL
);
